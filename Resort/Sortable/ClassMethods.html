def ordered
  ordered_elements = []
  elements = {}

  scoped.each do |element|
    if element.first?
      ordered_elements << element
    else
      elements[element.id] = element
    end
  end

  elements.length.times do
    ordered_elements << elements[ordered_elements.last.next_id]
  end
  ordered_elements
end
